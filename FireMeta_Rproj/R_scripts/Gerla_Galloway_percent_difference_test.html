Gerla_Galloway_percent_difference_test

2023-10-03

The purpose of this script is to calculate the percent difference by concentration and by the load for one study to see how they differ. Gerla and Galloway have the most complete dataset to do this test

library(ggplot2)
conc <- read_excel(here("inputs", "Studies", "Gerla_Galloway_1998", "Gerla_Galloway_1998.xlsx")) 

transport <- read_excel(here("inputs", "Studies", "Gerla_Galloway_1998", "Gerla_Galloway_1998_load.xlsx")) 
MeanConcTest <- conc %>%
  group_by(Climate) %>% 
  summarize(Ne = sum(Burn_Unburn == "Burn"),
                   Nc = sum(Burn_Unburn == "Unburn"),
                   Xe = mean(NO3[Burn_Unburn == "Burn"], na.rm = TRUE),
                   Xc = mean(NO3[Burn_Unburn == "Unburn"], na.rm = TRUE)) # taking the sum of observations for the unburn site and the burn site. It is also taking the mean of all of those observations

ggplot(conc, aes(x = Burn_Unburn, y = NO3, fill = Burn_Unburn)) +
  geom_boxplot() +
  scale_fill_manual(values=c("red3", "springgreen4")) +
  ggtitle("Concentration (mg_N_L)") +
  theme_bw()

# calculating the percent change 
percent_change_conc <- MeanConcTest %>% 
  group_by(Climate) %>% 
  mutate(percent_change1 = (((Xe-Xc)/Xc)*100))
MeanTransportTest <- transport %>%
  group_by(Climate, NO3_unit) %>% 
  summarize(Ne = sum(Burn_Unburn == "Burn"),
                   Nc = sum(Burn_Unburn == "Unburn"),
                   Xe = mean(NO3[Burn_Unburn == "Burn"], na.rm = TRUE),
                   Xc = mean(NO3[Burn_Unburn == "Unburn"], na.rm = TRUE)) # taking the sum of observations for the unburn site and the burn site. It is also taking the mean of all of those observations
## `summarise()` has grouped output by 'Climate'. You can override using the
## `.groups` argument.
ggplot(transport, aes(x = Burn_Unburn, y = NO3, fill = Burn_Unburn)) +
  geom_boxplot() +
  scale_fill_manual(values=c("red3", "springgreen4")) +
  ggtitle("Transport (mmol NO3_ha_s)") +
  theme_bw()

# calculating the percent change 
percent_change_transport <- MeanTransportTest %>% 
  group_by(Climate) %>% 
  mutate(percent_change1 = (((Xe-Xc)/Xc)*100))
combined_pc <- full_join(percent_change_conc, percent_change_transport)
## Joining with `by = join_by(Climate, Ne, Nc, Xe, Xc, percent_change1)`
ggplot(combined_pc, aes(x = percent_change1, y = Climate, color = NO3_unit)) +
  geom_point() +
  scale_color_manual(values = c("#00AFBB"),
                     guide = guide_legend(title = "Method"),
                     labels = c('Transport (mmol NO3_ha_s)', 'Concentration (mg_N_L)')) +
  xlab("Percent Change") +
  theme_bw()
